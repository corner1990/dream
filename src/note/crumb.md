##10. 时间格式化

	function dateFormat(t){        // t 是以秒为单位的值。
		var h = ~~(t/3600),        // t除以3600，取整，得到的就是小时。
		    m = ~~(t%3600/60),    // t求余3600，取模，得到的就是去除小时剩下的秒数（分钟 + 秒），再除以60，取整，得到的就是分钟。
		    s = ~~(t%3600%60);    // t求余3600，再求余60，剩下的自然就是“秒数”。

		 return h+'小时'+m+'分'+s+'秒';
	}
